extern void object::Tasks() {}

public void object::performTask(int task, float[] params) {
	if (task == 1) {
		collect(params);
	} else {
		message("Unrecognized task code: " + task, DisplayError);
	}
}

public void object::collect(float[] params) {
	//CONSTANTS
	float resource = params[0];
	point dropOff = new point(params[1], params[2]);
	bool shouldWait = (params[3] != 0);
	
	Bot me(this);
	me.hoverHeight = 100;
	
	while(me.get(resource, dropOff, 30, shouldWait)) {
		me.deposit(dropOff);
	}
	
	message("No more!");
}
