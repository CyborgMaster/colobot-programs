void addGoals(CommandCenter command) {
	command.addGoal("load", TrackedGrabber, integer(BlackBox)+",1");
	command.addGoal("board", WheeledGrabber, "");
}

extern void object::P1L3() {
	CommandCenter command = new CommandCenter();

	object that = this;
	command.registerSlave(that);
	//command.checkForUnregisteredSlaves(that);

	command.resetGoals();
	addGoals(command);

	//run slave script
	slave();
}
