extern void object::P1L3() {
	CommandCenter command = new CommandCenter();
	
	research("Tracked Bots");
	buildBot(TrackedGrabber, "Tracked Grabber");
	
	//wait for slave script to be run
	SlaveInfo slave;
	while(true) {
		slave = command.getFreeSlave(TrackedGrabber);
		if (slave != null) break;
		wait(1);
	}
	
	//command the slave to get the black box
	object o = radar(SpaceShip);
	slave.params[0] = BlackBox; //category
	slave.params[1] = o.position.x; //drop x
	slave.params[2] = o.position.y; //drop y
	slave.params[3] = 0; //shouldn't wait;
	slave.command = 1; //collect
	
	//go to the ship
	goto(o.position);
	
	//wait for the slave to finish
	while (slave.status == slave.BUSY) {
		wait(1);
	}
	
	message("Board spaceship and take off!");
}
