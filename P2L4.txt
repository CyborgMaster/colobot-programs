void addGoals(CommandCenter command) {
	command.addGoal("load", WingedGrabber, integer(BlackBox)+",1");
}

extern void object::P2L4() {
	CommandCenter command = new CommandCenter();

	object that = this;
	command.registerSlave(that);
	command.checkForUnregisteredSlaves(that);

	command.resetGoals();
	addGoals(command);

	//run slave script
	slave();
}
