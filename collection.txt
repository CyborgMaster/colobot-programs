extern void object::collect() {
	//CONSTANTS
	int resource =  Titanium;
	int flag = BlueFlag;
	bool shouldWait = true;
	
	Bot me(this);
	
	object o = radar(flag);
	me.bot_goto(o.position, 1);
	
	while(me.get(resource, 30, shouldWait)) {
		me.deposit(flag);
	}
	
	message("No more!");
}
