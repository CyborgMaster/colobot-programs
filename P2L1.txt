void addGoals(CommandCenter command) {
	Goal g = new LoadGoal(TitaniumOre, 4);
	command.addGoal(g);
	g = new BoardGoal(WheeledGrabber);
	command.addGoal(g);
}

extern void object::P2L1() {
	CommandCenter command = new CommandCenter();

	object that = this;
	command.registerSlave(that);
	command.checkForUnregisteredSlaves(that);

	command.resetGoals();
	addGoals(command);

	//run slave script
	slave();
}
