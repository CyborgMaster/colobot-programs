extern void object::Kill()
{
	int alienType = AlienAnt;
	
	Bot me(this);
	
	object item;
	
	aim(-20);
	
	while (true)
	{
		item = radar(alienType);
		me.follow(item, 30);
		motor(1,1);
		if (item != null) {
			turn(direction(item.position));
			motor(1,1);
			fire(1);
			wait(1.5);
		}
		wait(0.1);
	}
}
