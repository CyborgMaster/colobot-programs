extern void object::Slave() 
{
	CommandCenter command = new CommandCenter();
	object myThis = this;
	SlaveInfo info = command.registerSlave(myThis);
	
	int task;
	while(true) {
		task = info.command;
		if (task >= 0) break;
		wait(1); 
	}
	
	info.command = 0;
	performTask(task, info.params);
}
